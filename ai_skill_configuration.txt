/* Appian Integration Constants */
cons!AI_SKILL_RESUME_ANALYZER = "ai-skill-resume-job-matcher"
cons!AI_SKILL_TIMEOUT = 30 /* seconds */
cons!AI_SKILL_MAX_RETRIES = 3
cons!AI_SKILL_CACHE_DURATION = 300 /* 5 minutes */

/* Expression Rule: CALL_AI_SKILL_ResumeAnalysis */
rule!CALL_AI_SKILL_ResumeAnalysis(
  skillId: ri!skillId,
  input: ri!input
)

a!localVariables(
  local!startTime: now(),
  
  /* Validate input parameters */
  local!validation: rule!VALIDATE_AISkillInput(ri!input),
  
  if(
    local!validation.isValid,
    /* Proceed with AI Skill call */
    a!localVariables(
      local!aiResult: a!aiSkill(
        skillId: ri!skillId,
        input: ri!input,
        timeout: cons!AI_SKILL_TIMEOUT
      ),
      
      /* Process and validate AI response */
      local!processedResult: rule!PROCESS_AISkillResponse(
        response: local!aiResult,
        inputData: ri!input,
        processingTime: interval(local!startTime, now())
      ),
      
      /* Cache successful results */
      if(
        local!processedResult.success,
        rule!CACHE_AISkillResult(
          key: rule!GENERATE_CacheKey(ri!input),
          result: local!processedResult,
          duration: cons!AI_SKILL_CACHE_DURATION
        ),
        {}
      ),
      
      local!processedResult
    ),
    
    /* Return validation error */
    {
      success: false,
      recommendations: {},
      candidateAnalysis: {},
      metadata: {},
      errorDetails: local!validation.errorMessage
    }
  )
)

/* Performance Monitoring Rules */
rule!LOG_AISkillPerformance(
  skillId: ri!skillId,
  processingTime: ri!processingTime,
  success: ri!success,
  inputSize: ri!inputSize,
  outputSize: ri!outputSize
)

/* This logs performance metrics for monitoring and optimization */
rule!WRITE_TO_DataStore(
  dataStore: cons!DS_AI_SKILL_METRICS,
  entity: "AISkillUsage",
  data: {
    skillId: ri!skillId,
    timestamp: now(),
    processingTimeMs: ri!processingTime,
    success: ri!success,
    inputSizeBytes: ri!inputSize,
    outputSizeBytes: ri!outputSize,
    userId: loggedInUser()
  }
)

/* Error Handling and Retry Logic */
rule!RETRY_AISkillCall(
  skillId: ri!skillId,
  input: ri!input,
  maxRetries: ri!maxRetries: cons!AI_SKILL_MAX_RETRIES
)

a!localVariables(
  local!attempts: 0,
  local!success: false,
  local!result: {},
  
  /* Retry loop */
  while(
    and(local!attempts < ri!maxRetries, not(local!success)),
    a!localVariables(
      local!attempts: local!attempts + 1,
      local!currentResult: rule!CALL_AI_SKILL_ResumeAnalysis(
        skillId: ri!skillId,
        input: ri!input
      ),
      local!success: local!currentResult.success,
      local!result: local!currentResult,
      
      /* Wait before retry if failed */
      if(
        and(not(local!success), local!attempts < ri!maxRetries),
        rule!WAIT(seconds: local!attempts * 2), /* Exponential backoff */
        {}
      )
    )
  ),
  
  /* Add retry information to result */
  local!result & {
    metadata: local!result.metadata & {
      retryAttempts: local!attempts,
      finalAttempt: local!success
    }
  }
)
