/* Appian Expression Rule: JOB_SubmitApplication
   Handles the job application submission process */

rule!JOB_SubmitApplication(
  jobId: ri!jobId,
  candidateId: ri!candidateId,
  applicationSource: "AI_RECOMMENDATION"
)

a!localVariables(
  /* Get job and candidate details */
  local!jobDetails: rule!GET_JobDetails(ri!jobId),
  local!candidateProfile: rule!GET_CandidateProfile(ri!candidateId),
  
  /* Validate application eligibility */
  local!validationResult: rule!VALIDATE_ApplicationEligibility(
    jobId: ri!jobId,
    candidateId: ri!candidateId
  ),
  
  if(
    local!validationResult.isEligible,
    /* Proceed with application */
    a!localVariables(
      /* Create application record */
      local!applicationId: rule!CREATE_JobApplication(
        jobId: ri!jobId,
        candidateId: ri!candidateId,
        applicationDate: now(),
        source: ri!applicationSource,
        status: "SUBMITTED",
        documents: rule!GET_CandidateDocuments(ri!candidateId),
        additionalInfo: {
          aiRecommended: true,
          matchScore: rule!GET_JobMatchScore(ri!jobId, ri!candidateId),
          applicationChannel: "SELF_SERVICE_PORTAL"
        }
      ),
      
      /* Trigger workflow processes */
      local!workflowResults: {
        /* Start HR review process */
        rule!START_HRReviewProcess(local!applicationId),
        
        /* Send confirmation notifications */
        rule!SEND_ApplicationConfirmation(
          candidateId: ri!candidateId,
          jobId: ri!jobId,
          applicationId: local!applicationId
        ),
        
        /* Update candidate activity log */
        rule!LOG_CandidateActivity(
          candidateId: ri!candidateId,
          activity: "JOB_APPLICATION_SUBMITTED",
          details: {
            jobId: ri!jobId,
            applicationId: local!applicationId,
            timestamp: now()
          }
        ),
        
        /* Notify hiring manager if urgent */
        if(
          local!jobDetails.urgency = "HIGH",
          rule!NOTIFY_HiringManager(
            jobId: ri!jobId,
            applicationId: local!applicationId,
            candidateId: ri!candidateId,
            priority: "HIGH"
          ),
          null
        )
      },
      
      /* Return success response */
      {
        success: true,
        applicationId: local!applicationId,
        message: "Your application has been successfully submitted for " & local!jobDetails.jobTitle,
        nextSteps: {
          "HR will review your application within 3-5 business days",
          "You will receive email updates on your application status", 
          "Complete any additional assessments if required",
          "Prepare for potential interview rounds"
        },
        estimatedTimeToResponse: rule!CALCULATE_ResponseTime(ri!jobId),
        trackingUrl: rule!GENERATE_ApplicationTrackingUrl(local!applicationId)
      }
    ),
    
    /* Return validation error */
    {
      success: false,
      message: local!validationResult.errorMessage,
      suggestions: local!validationResult.suggestions
    }
  )
)

/* Supporting Rule: VALIDATE_ApplicationEligibility */
rule!VALIDATE_ApplicationEligibility(
  jobId: ri!jobId,
  candidateId: ri!candidateId
)

a!localVariables(
  local!existingApplication: rule!CHECK_ExistingApplication(ri!jobId, ri!candidateId),
  local!jobStatus: rule!GET_JobStatus(ri!jobId),
  local!candidateStatus: rule!GET_CandidateStatus(ri!candidateId),
  
  {
    isEligible: and(
      isnull(local!existingApplication),
      local!jobStatus = "ACTIVE",
      local!candidateStatus = "ACTIVE",
      not(rule!IS_CandidateBlacklisted(ri!candidateId))
    ),
    errorMessage: if(
      not(isnull(local!existingApplication)),
      "You have already applied for this position",
      if(
        local!jobStatus <> "ACTIVE",
        "This job posting is no longer active",
        if(
          local!candidateStatus <> "ACTIVE",
          "Your candidate profile needs to be activated",
          if(
            rule!IS_CandidateBlacklisted(ri!candidateId),
            "Application not permitted at this time",
            "Application validation failed"
          )
        )
      )
    ),
    suggestions: {
      "Review similar open positions",
      "Update your candidate profile",
      "Contact HR for assistance"
    }
  }
)
